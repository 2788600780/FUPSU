\subsection{Invertible Bloom Lookup Tables}

Invertible Bloom Lookup Tables (IBLTs) were introduced by Goodrich and
Mitzenmacher~\cite{GM11}. An IBLT is a variant of Bloom filter that supports,
in addition to inserting and deleting elements, the ability to \emph{list} all
stored elements. Unlike a standard Bloom filter---which only answers membership
queries with a configurable false-positive rate and requires $\approx 60\times$
expansion for negligible error---an IBLT stores key-value pairs in its cells
and achieves listing with expansion factor $1.5$--$4.5\times$ (for $k = 4$ hash
functions). This makes IBLTs uniquely suited for set reconciliation and private
set union, where the goal is to recover the entire set rather than answer
individual membership queries.

An IBLT is parameterized by $\mathsf{prm} = (M, k, \ell, H)$: a modulus $M \in
\N_{\geq 2}$ (typically $2^{64}$), $k$ hash functions $H = \{h_i : \U \to
[\ell]\}_{i \in [k]}$ each mapping to a distinct subtable of size $\ell$, and
two $k \times \ell$ matrices $\mathsf{sum}$ (values in $\Z_M$) and
$\mathsf{cnt}$ (non-negative integers). A cell with $\mathsf{cnt}[i][j] = 1$
is a \emph{singleton}: its sum equals the unique element stored there.

IBLT encoding is \emph{linear}: $\Encode(A \cup B) = \Encode(A) \oplus
\Encode(B)$ and $\Encode(A \setminus B) = \Encode(A) \ominus \Encode(B)$,
where $\oplus$/$\ominus$ denote cell-wise modular addition/subtraction.
The three core operations are given in
Algorithms~\ref{alg:encode}--\ref{alg:list}.

\begin{algorithm}[ht]
\caption{$\Encode_{\mathsf{prm}}(X)$ --- IBLT encoding.}
\label{alg:encode}
\begin{algorithmic}
\STATE \textbf{Input:} $\mathsf{prm} = (M, k, \ell, H)$, set $X \subseteq \U$, empty IBLT.
\STATE \textbf{Output:} $\mathsf{IBLT} = (\mathsf{sum}, \mathsf{cnt}, \mathsf{prm})$ containing all $x \in X$.
\STATE Initialize $\mathsf{sum}[i][j] \leftarrow 0$, $\mathsf{cnt}[i][j] \leftarrow 0$ for all $i \in [k], j \in [\ell]$.
\FOR{each $x \in X$}
  \FOR{$i \in [k]$}
    \STATE $j \leftarrow h_i(x)$
    \STATE $\mathsf{sum}[i][j] \leftarrow \mathsf{sum}[i][j] + x \pmod M$
    \STATE $\mathsf{cnt}[i][j] \leftarrow \mathsf{cnt}[i][j] + 1$
  \ENDFOR
\ENDFOR
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[ht]
\caption{$\DeleteSet_{\mathsf{prm}}(\mathsf{IBLT}, X)$ --- batch deletion.}
\label{alg:deleteset}
\begin{algorithmic}
\STATE \textbf{Input:} $\mathsf{IBLT}$ and set $X \subseteq \U$.
\STATE \textbf{Output:} Updated $\mathsf{IBLT}$ with all $x \in X$ removed.
\FOR{each $x \in X$}
  \FOR{$i \in [k]$}
    \STATE $j \leftarrow h_i(x)$
    \STATE $\mathsf{sum}[i][j] \leftarrow \mathsf{sum}[i][j] - x \pmod M$
    \STATE $\mathsf{cnt}[i][j] \leftarrow \mathsf{cnt}[i][j] - 1$
  \ENDFOR
\ENDFOR
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[ht]
\caption{$\List(\mathsf{IBLT})$ --- full listing (peeling).}
\label{alg:list}
\begin{algorithmic}
\STATE \textbf{Input:} $\mathsf{IBLT} = (\mathsf{sum}, \mathsf{cnt}, \mathsf{prm})$.
\STATE \textbf{Output:} Set $V$ of all encoded elements, or $\bot$ on failure.
\STATE $V \leftarrow \emptyset$
\WHILE{there exists $(i,j)$ with $\mathsf{cnt}[i][j] = 1$}
  \STATE $x \leftarrow \mathsf{sum}[i][j]$
  \STATE $V \leftarrow V \cup \{x\}$
  \STATE $\DeleteSet_{\mathsf{prm}}(\mathsf{IBLT}, \{x\})$
\ENDWHILE
\IF{$\mathsf{cnt}[i][j] = 0$ for all $(i,j)$}
  \STATE \textbf{output} $V$
\ELSE
  \STATE \textbf{output} $\bot$
\ENDIF
\end{algorithmic}
\end{algorithm}

\begin{theorem}[IBLT Listing~\cite{GM11,PT26}]\label{thm:iblt}
For $m = k\ell > (c_k + \varepsilon)\tau$ with $n \leq \tau$ elements,
$\Pr[\List\text{ fails}] = O(\tau^{-k+2})$. With $k = 4$, expansion
$m/\tau \in [1.5, 4.5]$, failure $\leq 2^{-\lambda}$.
\end{theorem}

If $\List$ terminates with non-singleton cells remaining (a $2$-core),
listing has failed. The expansion factor $e = m/n$ must exceed $c_k$ to
guarantee an empty $2$-core with overwhelming probability.

\subsection{Union Peelability}

Piske and Trieu~\cite{PT26} observed that the union of two IBLTs can be peeled
without materializing the combined table. For $\mathsf{IBLT}_0 =
\Encode(X_0)$ and $\mathsf{IBLT}_1 = \Encode(X_1)$, the virtual union is
$\mathsf{IBLT}_\cup = \mathsf{IBLT}_0 \oplus \mathsf{IBLT}_1$.
The key primitive is the \emph{union peel} function $\uPeel$, defined below.

\begin{definition}[$\uPeel$~\cite{PT26}]\label{def:upeel}
\footnotesize
\[
\uPeel(\mathsf{IBLT}_0, \mathsf{IBLT}_1, i, j) = \begin{cases}
\mathsf{sum}_0[i][j], & \mathsf{cnt}_0{=}1 \land \mathsf{cnt}_1{=}0 \\
\mathsf{sum}_1[i][j], & \mathsf{cnt}_0{=}0 \land \mathsf{cnt}_1{=}1 \\
\mathsf{sum}_0[i][j], & \mathsf{cnt}_0{=}\mathsf{cnt}_1{=}1
                       \land \mathsf{sum}_0{=}\mathsf{sum}_1 \\
\bot, & \text{otherwise}
\end{cases}
\]
\end{definition}

Intuitively, $\uPeel$ distinguishes three cases: the peeled element is unique
to $P_0$, unique to $P_1$, or shared with equal sums in both IBLTs.
Secure evaluation of $\uPeel$ uses OT for sum retrieval and OPRF for
equality comparison, as specified by the $\Pi_{\mathsf{UnionPeel}}$ protocol.

\begin{theorem}[Union Peel Equivalence~\cite{PT26}]\label{thm:upeel}
For union-peelable IBLTs: $\Peel(\mathsf{IBLT}_\cup, i, j) =
\uPeel(\mathsf{IBLT}_0, \mathsf{IBLT}_1, i, j)$.
\end{theorem}

\begin{center}
\fbox{\begin{minipage}{0.95\columnwidth}
\footnotesize
\smallskip
\noindent\textbf{Protocol: Union Peel}

\smallskip\noindent\textbf{Parameters:}
Two parties $P_0, P_1$; IBLT parameters $\mathsf{prm} = (M, k, \ell, H)$;
a public query set $Q \subseteq [k] \times [\ell]$;
$\cF_{\mathsf{OT}}$ (1-out-of-2, 1-out-of-3) and $\cF_{\mathsf{OPRF}}$.

\smallskip\noindent\textbf{Input:}
Each $P_b$ provides $\mathsf{IBLT}_b = (\mathsf{sum}_b, \mathsf{cnt}_b)$ with parameters $\mathsf{prm}$.

\smallskip\noindent\textbf{Protocol:}
\begin{enumerate}
\item \textbf{[Sum retrieval.]}
For each $(i,j) \in Q$, $P_0$ and $P_1$ run 1-out-of-2 OT:
$P_0$ inputs $c_{i,j} = \mathbf{1}[\mathsf{cnt}_0[i][j] = 0]$;
$P_1$ inputs $m_0 = \bot$ and $m_1 = \mathsf{sum}_1[i][j]$ if $\mathsf{cnt}_1[i][j]=1$ else $\bot$.
$P_0$ obtains $f_{i,j}$.

\item \textbf{[OPRF evaluation.]}
For each $(i,j) \in Q$, $P_1$ queries $\cF_{\mathsf{OPRF}}$ on
$z_{i,j} = \mathsf{sum}_1[i][j]$ (if $\mathsf{cnt}_1[i][j] = 1$, else $\bot$),
receiving $y_{i,j}$; $P_0$ receives PRF key $k_{i,j}$.
$P_0$ then computes
$u_{i,j} = F_{k_{i,j}}(\mathsf{sum}_0[i][j])$ if $\mathsf{cnt}_0[i][j] = 1$;
$F_{k_{i,j}}(f_{i,j})$ if $\mathsf{cnt}_0[i][j] = 0 \land f_{i,j} \neq \bot$;
a random $r_{i,j}$ otherwise.

\item \textbf{[Result delivery.]}
For each $(i,j) \in Q$, $P_0$ and $P_1$ run 1-out-of-3 OT.
$P_1$ inputs $d_{i,j} = \mathsf{cnt}_1[i][j]$ (if $\in \{0,1\}$, else $2$).
$P_0$ inputs $w_{i,j} = (w_0, u_{i,j}, \bot)$ where
$w_0 = \mathsf{sum}_0[i][j]$ if $\mathsf{cnt}_0[i][j] = 1$ else $\bot$.
$P_1$ receives $g_{i,j}$.

\item \textbf{[Output.]}
$P_1$ locally computes
$v_{i,j} = \mathsf{sum}_1[i][j]$ if $d_{i,j}{=}1 \land g_{i,j}{=}y_{i,j}$;
$\bot$ if $d_{i,j}{=}1 \land g_{i,j}{\neq}y_{i,j}$; $g_{i,j}$ otherwise.
$P_1$ sends $V = \{(v_{i,j}, i, j)\}$ to $P_0$. Both output $V$.
\end{enumerate}
By Theorem~\ref{thm:upeel}, the non-$\bot$ elements in $V$ equal those
recoverable from the virtual union $\mathsf{IBLT}_0 \oplus \mathsf{IBLT}_1$.

\smallskip
\end{minipage}}
\captionof{figure}{The Union Peel protocol~\cite{PT26}.}
\label{fig:unionpeel}
\end{center}

\begin{theorem}[Union Peel Security~\cite{PT26}]\label{thm:unionpeel-sec}
$\Pi_{\mathsf{UnionPeel}}$ UC-realizes $\cF_{\mathsf{UnionPeel}}$
in the $(\cF_{\mathsf{OPRF}}, \cF_{\mathsf{OT}})$-hybrid model against
semi-honest adversaries, achieving static PSU on $2^{20}$ elements in
under 3~seconds.
\end{theorem}

\subsection{Cryptographic Primitives}

\textbf{Symmetric encryption.} We use a symmetric encryption scheme
$\SE = (\SE.\Enc, \SE.\Dec)$ with 256-bit keys, instantiated with
ChaCha20~\cite{Ber08}. The IBLT ciphertexts are encrypted under per-epoch
keys derived from the PRF chain (Section~\ref{sec:protocol}).

\textbf{PRF.} We use HMAC-SHA256~\cite{BCK96} as a secure PRF:
$F: \{0,1\}^\kappa \times \{0,1\}^* \to \{0,1\}^\kappa$. Key evolution uses a
fixed domain separator: $K_t \leftarrow F(K_{t-1}, \text{``evolve''})$.

\begin{lemma}[PRF Key One-Wayness]\label{lem:prf-ow}
If $F$ is a secure PRF, then for any PPT $\cA$ and fixed $c$:
$\Pr[\cA(F(K, c)) = K : K \sample \{0,1\}^\kappa] \leq \negl(\kappa)$.
\end{lemma}

\textbf{VOLE and OT extension.} Vector Oblivious Linear Evaluation (VOLE)
enables two parties to obtain additive shares of $\Delta \cdot \vec{x}$,
realizable from $\kappa$ base OTs with sublinear communication~\cite{RS21,RR22};
silent VOLE~\cite{RRT23} eliminates the base OTs entirely. The VOLE state is
reusable across all epochs.

\textbf{Batched OPRF.} The KKRT16/RS21/RR22 framework~\cite{KKRT16,RS21,RR22,HKL+24}
constructs batched OPRF from VOLE and OKVS.

We now define the three ideal functionalities used in our protocol.

The 1-out-of-2 OT functionality $\cF_{\mathsf{OT}}$ allows a receiver to
obtain one of two messages held by a sender, without the sender learning
which message was chosen.

\subsection{UC Framework}

We use the Universal Composability framework~\cite{Can01} with adaptive
corruptions. A protocol $\Pi$ UC-realizes an ideal functionality $\cF$ if for
every PPT adversary $\cA$ there exists a PPT simulator $\cS$ such that no PPT
environment $\cZ$ can distinguish the real execution from the ideal execution.
We operate in the $(\cF_{\mathsf{OPRF}}, \cF_{\mathsf{OT}})$-hybrid model.
The ideal functionality for our protocol, $\cF_{\mathsf{updPSU}}$, is specified
in Section~\ref{sec:protocol}.
